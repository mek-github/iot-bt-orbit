rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Helper function to check if user is authenticated
    function isSignedIn() {
      return request.auth != null;
    }
    
    // Helper function to check if user is the owner
    function isOwner(userId) {
      return isSignedIn() && request.auth.uid == userId;
    }
    
    // Users collection
    match /users/{userId} {
      // Anyone authenticated can read user profiles
      allow read: if isSignedIn();
      // Users can only write their own profile
      allow write: if isOwner(userId);
    }
    
    // Events collection
    match /events/{eventId} {
      // Anyone authenticated can read events
      allow read: if isSignedIn();
      
      // Anyone authenticated can create events
      allow create: if isSignedIn();
      
      // Allow updates if:
      // 1. User is the host of the event, OR
      // 2. Any authenticated user (for checkedInCount updates during check-in/out)
      allow update: if isSignedIn();
      
      // Only host can delete their own events
      allow delete: if isSignedIn() && resource.data.hostId == request.auth.uid;
      
      // Check-ins subcollection
      match /checkins/{userId} {
        // Anyone authenticated can read check-ins
        allow read: if isSignedIn();
        
        // Users can only create/delete their own check-ins
        allow create, delete: if isOwner(userId);
      }
    }
  }
}
